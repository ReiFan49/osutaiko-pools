## Fun Snippet

def rateSD(od, rate=3):
   newSD = taikoSD(od) * 3 / rate
   if newSD >= 5:
     return (newSD - 5) / 0.5 + 5
   else:
     return (newSD - 3) / 0.4

dtSD = lambda od: rateSD(od, 4)
htSD = lambda od: rateSD(od, 2)
